function getRandomFromList(items: string[] = [], excludeValue = ""): string {

  const safeItems = items.filter(Boolean);

  if (safeItems.length === 0) return "";

  const candidates = safeItems.filter((item) => item !== excludeValue);

  const pool = candidates.length > 0 ? candidates : safeItems;

  return pool[Math.floor(Math.random() * pool.length)] ?? "";

}

  const selection = window.getSelection();

  const range = document.createRange();

  range.selectNodeContents(el);

  selection?.removeAllRanges();

  selection?.addRange(range);

  let success = false;

  try { success = document.execCommand('copy'); } catch (err) { console.error(err); }

const compressImage = (base64Str: string): Promise<string> => {

  return new Promise((resolve) => {

    const img = new Image();

    img.crossOrigin = "Anonymous";

    img.onload = () => {

      const canvas = document.createElement('canvas');

      const MAX_WIDTH = 800;

      let width = img.width;

      let height = img.height;

      if (width > MAX_WIDTH) {

        height = Math.round((height * MAX_WIDTH) / width);

        width = MAX_WIDTH;

      }

      canvas.width = width;

      canvas.height = height;

      const ctx = canvas.getContext('2d');

     

      if(ctx) {

          ctx.fillStyle = "#FFFFFF";

          ctx.fillRect(0, 0, width, height);

          ctx.drawImage(img, 0, 0, width, height);

      }

     

      resolve(canvas.toDataURL('image/jpeg', 0.85));

    };

    img.onerror = () => {

        console.warn("Gagal mengompres gambar, menggunakan versi asli.");

        resolve(base64Str);

    };

    img.src = base64Str;

  });

};

    }

  let currentKey = trimmedImageKey || apiKey;

 

  let endpointsToTry = [];

  if (currentKey.startsWith("sk-") && !currentKey.startsWith("sk-or-")) {

      endpointsToTry.push({ type: 'openai', key: currentKey });

  } else if ((currentKey.length === 64 && /^[0-9a-fA-F]+$/.test(currentKey)) || currentKey.startsWith("sk-or-")) {

      currentKey = apiKey;

  }

  endpointsToTry.push({ type: 'imagen-object', key: currentKey });

  endpointsToTry.push({ type: 'imagen-array', key: currentKey });

  endpointsToTry.push({ type: 'gemini-flash', key: currentKey });

      try {

          if (endpoint.type === 'openai') {

              const res = await fetch("https://api.openai.com/v1/images/generations", {

                  method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${endpoint.key}` },

                  signal

              });

              const base64 = data.candidates?.[0]?.content?.parts?.find((p: any) => p.inlineData)?.inlineData?.data;

      }

  }

}

const sanitizeContentMap = (map: Record<string, string[]>) => {

  const cleanMap: Record<string, string[]> = {};

  if (!map) return cleanMap;

  for (const key in map) {

     const arr = map[key] || [];

     const safeArr = [];

     for (let i = 0; i < arr.length; i++) {

        safeArr.push(arr[i] === undefined || arr[i] === null ? "" : arr[i]);

     }

     cleanMap[key] = safeArr;

  }

  return cleanMap;

};
